package com.jiro4989.tkfm.model

import javafx.beans.property.DoubleProperty
import javafx.beans.property.SimpleDoubleProperty

class RectangleModel {
  private val width = SimpleDoubleProperty(0.0)
  private val height = SimpleDoubleProperty(0.0)

  constructor (w: Double, h: Double) {
    setWidth(w)
    setHeight(h)
  }

  fun widthProperty() = width
  fun heightProperty() = height
  fun getWidth() = width.get()
  fun getHeight() = height.get()

  fun setWidth(w: Double) {
      width.set(w)
  }

  fun setHeight(h: Double) {
      height.set(h)
  }
}
